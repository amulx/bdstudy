又名 控制反转 降低耦合度

  • __Construct()
  • call_user_func_array

42class Container{
43    protected $binds;
44    protected $instances;
45
46    public  function bind($abstract,$concrete){
47        if ($concrete instanceof Closure) { // 判断是不是闭包化
48            $this->binds[$abstract] = $concrete;
49        } else {
50            $this->instances[$abstract] = $concrete;// 正常实例化
51        }
52    }
53
54    public function make($abstract,$parameters=[]){
55        if (isset($this->instances[$abstract])) {
56            return $this->instances[$abstract];
57        }
58        array_unshift($parameters, $this);
59        return call_user_func_array($this->binds[$abstract], $parameters);
60    }
61}

64$dog = new dog();
